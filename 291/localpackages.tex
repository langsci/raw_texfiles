\newglossarystyle{audersetglosses}% Following a solution by Nicola Talbot (https://tex.stackexchange.com/a/325566)
{%
    \setglossarystyle{mcolalttree}
    % remove empty line
    % from https://tex.stackexchange.com/questions/385883/delete-empty-line-in-glossaries
    \renewenvironment{theglossary}%
	    {%
	        \begin{multicols}{2}%
	            \raggedright%
	            \def\@gls@prevlevel{-1}%
	    }%
	    {\par\end{multicols}}%
	\renewcommand*{\glstreenamefmt}[1]{##1}%
	\renewcommand*{\glossaryentrynumbers}[1]{\striprelax##1\endstriprelax}%
	\renewcommand*{\glsXchapterXglsnumberformat}[2]{##2}%
	\renewcommand*{\delimR}{,}%
	% Include conditional, following a recipe by Nicola Talbot.
	\renewcommand{\glossentry}[2]{%
		\edef\doifinlocation{\noexpand\ifinlocation{\thechapter}{##2}}%
		\doifinlocation
		{% Copied from the mcols-alttree style
			\ifnum\@gls@prevlevel=0\relax
		    \else
		       \settowidth{\glstreeindent}{\glstreenamefmt{\@glswidestname\space}}%
		    \fi
		    \hangindent\glstreeindent
		    \parindent\glstreeindent
		    \makebox[0pt][r]{\glstreenamebox{\glstreeindent}{%
		       \glsentryitem{##1}\glstreenamefmt{\glstarget{##1}{\glossentryname{##1}}}}}%
		    \ifglshassymbol{##1}{(\glossentrysymbol{##1})\space}{}%
		    \glossentrydesc{##1}\glspostdescription\\
		    \def\@gls@prevlevel{0}%
		}%
	}%
}
