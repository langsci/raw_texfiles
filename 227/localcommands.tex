	\forestset{
		sn edges/.style={for tree={parent anchor=south, child anchor=north,align=center}},
		nice empty nodes/.style={
			for tree={calign=fixed edge angles},
			delay={where content={}{shape=coordinate,for parent={for children={anchor=north}}}{}} },
		%Code by cfr; https://tex.stackexchange.com/a/281546
		my nice empty nodes/.style={% modified from manual page 52
			for tree={
				calign=fixed edge angles,
				calign angle=60,
			},
			delay={
				where n children=0{
					if content={}{
						content=\strut,
						anchor=north,
					}{
						align=center
					},
				}{
					if content={}{
						shape=coordinate,
						for parent={
							for children={
								anchor=north
							}
						}
					}{}
				}
			},
		},
		my pretty nice empty nodes/.style={
			for tree={
				calign=fixed edge angles,
				calign angle=60,
				parent anchor=south,
				delay={
					where n children=0{
						if content={}{
							content=\strut,
							anchor=north,
						}{
							align=center
						},
					}{
						if content={}{
							inner sep=0pt,
							edge path={\noexpand\path [\forestoption{edge}] (!u.parent anchor) -- (.south)\forestoption{edge label};}
						}{}
					}
				}
			}
		}
	}
