% macro for unmarked footnote continued from previous page
\newcommand\blfootnote[1]{%
  \begingroup
  \renewcommand\thefootnote{}\footnote{\hspace*{-1.5em}#1}%
  \addtocounter{footnote}{-1}%
  \endgroup
}

\newcounter{TextPageNo}
\setcounter{TextPageNo}{2}

\newcommand{\TextPage}[3][!*!,!]{%
\newpage
\ifthenelse{\equal{#1}{!*!,!}}
{%\begin{center}
\chead{\upshape--- \arabic{TextPageNo} ---}
%\end{center}
}
{\ifthenelse{\equal{#1}{no}}{}{#1}}
\stepcounter{TextPageNo}
% % \setlength\parindent{2em}
\begin{paracol}{2}
\setcounter{footnote}{0}
{\selectlanguage{french}
  #2}
\switchcolumn
\setcounter{footnote}{0}
{\selectlanguage{english}
  #3}
\end{paracol}
}

\newcommand{\LargerTextPage}[3][!*!,!]{%
\newpage
\ifthenelse{\equal{#1}{!*!,!}}
{%\begin{center}
\chead{\upshape--- \arabic{TextPageNo} ---}
%\end{center}
}
{\ifthenelse{\equal{#1}{no}}{}{#1}}
\stepcounter{TextPageNo}
% % \setlength\parindent{2em}
\begin{paracol}{2}
\setcounter{footnote}{0}
{\noindent\begin{minipage}[b]{\linewidth}
\selectlanguage{french}
  #2\end{minipage}}
\switchcolumn
\setcounter{footnote}{0}
{\noindent\begin{minipage}[b]{\linewidth}
\selectlanguage{english}
  #3\end{minipage}}
\end{paracol}
}

%\end{center}
}}
{\ifthenelse{\equal{#1}{no}}{}{#1}}
\stepcounter{TextPageNo}
\setcounter{footnote}{0}
{\selectlanguage{french}
  #2}
}
